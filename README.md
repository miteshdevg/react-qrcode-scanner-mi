## Install

```bash
npm install --save react-qrcode-scanner-mi
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-qrcode-scanner-mi'
import 'react-qrcode-scanner-mi/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
